select p.product_name, i.code_product , c.name as category ,sum(i.quantity) as total_product from inventory as i join product as p on i.code_product = p.product_code join category as c on p.id_category = c.id WHERE i.code_warehouse = '1ef3ebc0-a1c0-6abc-b3b4-aabbccddeeff' GROUP by i.code_product ,p.product_name
